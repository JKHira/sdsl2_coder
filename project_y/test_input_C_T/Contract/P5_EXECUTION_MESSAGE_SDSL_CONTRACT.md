// @Structure.P5_C_MSEP_EMIDL_L5_STRATEGY_SIGNAL: L5 strategy signal message schema
struct StrategySignal {
  signal_id: SignalId,          // REQUIRED: UUID generated by Strategy Engine
  model_output_id: ModelOutputId?, // Optional: Link to inference
  instrument_id: s,             // e.g., "BTC-USD-PERP.HYPERLIQUID"
  direction: "BUY" | "SELL",
  strength: f,                  // 0.0 - 1.0 confidence
  suggested_size: f?,           // Optional: Strategy's size suggestion
  timestamp_ns: i64,            // UTC nanoseconds
  trace_id: s                   // Propagated trace context
}
// @Rule.P5_C_MSEP_TOPIC_STRATEGY_SIGNALS: @Structure.P5_C_MSEP_EMIDL_L5_STRATEGY_SIGNAL;

// @Structure.P5_C_BMOM_EMIDL_BM_EXECUTION_INTENT: L5/L6 boundary execution intent message schema
struct ExecutionIntent {
  intent_id: IntentId,          // REQUIRED: Generated by Execution Policy (= clordid)
  signal_id: SignalId,          // REQUIRED: Link to originating signal
  instrument_id: s,
  side: "BUY" | "SELL",
  order_type: "LIMIT" | "MARKET" | "STOP_LIMIT",
  quantity: Decimal,            // Exact quantity
  limit_price: Decimal?,        // Required for LIMIT/STOP_LIMIT
  stop_price: Decimal?,         // Required for STOP_LIMIT
  time_in_force: "GTC" | "IOC" | "FOK" | "GTD" = "GTC",
  timestamp_ns: i64,
  trace_id: s,
  tags: Map<s, s>               // strategy:system5, signal:{signal_id}, etc.
}
// @Rule.P5_C_BMOM_TOPIC_EXECUTION_INTENTS: @Structure.P5_C_BMOM_EMIDL_BM_EXECUTION_INTENT;

  timestamp_ns: i64,
  trace_id: s,
